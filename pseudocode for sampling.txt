subset getSubset(start, end)
{
	go to file

	while the COUNTER != start
	{
		COUNTER = next day
		stocks.getQueue.pop
	}
	
	subset<stockStats> ALLDATES
	
	do
	{
		ALLDATES.push(stocks.getQueue.head)

	} while (COUNTER != end)
	
	return ALLDATES
}
